A. 执行操作后的变量值#

  • 时间复杂度: $O(n)$
  • 空间复杂度: $O(1)$
class Solution {
public:
    int finalValueAfterOperations(vector<string>& operations) {
        int X = 0;    
        for (auto & s: operations) {
            if (s[1] == '+') X++; 
            else X--;
        }
        return X;
    }
};

B. 数组美丽值求和#

通过正反两次遍历记录每个值左边的最大值和右边的最小值.

  • 时间复杂度: $O(n)$
  • 空间复杂度: $O(n)$
class Solution {
public:
    int sumOfBeauties(vector<int>& nums) {
        int n = nums.size();
        vector<int> left(n, INT_MIN), right(n, INT_MAX);
        int ans = 0;
        left[0] = nums[0]; right[n - 1] = nums[n - 1];
        for (int i = 1; i < n; i++) left[i] = max(left[i-1], nums[i]);
        for (int i = n - 2; i >= 0; i--) right[i] = min(right[i + 1], nums[i]);
        for (int i = 1; i < n - 1; i++) {
            if (left[i - 1] < nums[i] && nums[i] < right[i + 1]) ans +=2;
            else if (nums[i - 1] < nums[i] && nums[i] < nums[i + 1]) ans++;
        }
        return ans;
    }
};

C. 检测正方形#

注意轴对齐正方形定义,可通过一个map来记录所有的坐标.

class DetectSquares {
    unordered_map<int, unordered_map<int, int>> mp;
public:
    DetectSquares() {
    }
    
    void add(vector<int> point) {
        mp[point[0]][point[1]]++;
    }
    
    int count(vector<int> point) {
        int x = point[0], y = point[1];
        int ans = 0;
        for (auto [y1, c]: mp[x]) {
            if (y1 == y) continue;
            int len = abs(y1 - y);
            ans += (mp[x + len][y1] * mp[x+len][y] + mp[x-len][y1] * mp[x-len][y]) * c;
        }
        return ans;
    }
};

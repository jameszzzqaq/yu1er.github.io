A: 差的绝对值为 K 的数对数目#

  • 时间复杂度:$O(n)$
  • 空间复杂度:$O(n)$
class Solution {
public:
    int countKDifference(vector<int>& nums, int k) {
        map<int, int> m;
        int ans;
        for (auto && x : nums) {
            ans += (m[x + k] + m[x - k]);
            m[x]++;
        }
        return ans;
    }
};

B: 从双倍数组中还原原数组#

两个关键点: 1. 元素个数一定为偶数 2. 结果数组元素个数一定是原数组二分之一

  • 时间复杂度: 排序: $O(nlgn)$ 遍历: $O(n)$
  • 空间复杂度: $O(n)$
class Solution {
public:
    vector<int> findOriginalArray(vector<int>& changed) {
        auto n = changed.size();
        if (n & 1) return {};
        sort(changed.begin(), changed.end());
        vector<int> ans;
        unordered_map<int, int> mp;
        for (auto && x : changed) {
            if (mp[x] == 0) {
                ans.push_back(x);
                mp[x * 2]++;
            }else {
                mp[x]--;
            }
        }
        if (ans.size() != (n >> 1)) return {};
        return ans;
    }
};

C: 租车的最大盈利#

动态规划, 类似背包问题,但要注意这是个区间,每次都要更新区间中的值.

  • 时间复杂度: $O(n * m)$
  • 空间复杂度: $O(n)$
class Solution {
    using LL = long long;
public:
    long long maxTaxiEarnings(int n, vector<vector<int>>& rides) {
        sort(begin(rides), end(rides));
        vector<LL> dp(n + 1);


        int i = 0;
        LL ans = 0;
        for (auto && ride: rides) {
            for ( ; i < ride[0]; i++) {
                dp[i + 1] = max(dp[i + 1], dp[i]);
            }
            dp[ride[1]] = max(dp[ride[1]], dp[ride[0]] + ride[1] - ride[0] + ride[2]);
        }
        for (; i < n; i++) {
            dp[i+1] = max(dp[i], dp[i+1]);
        }
        for (auto && x : dp) {
            ans = max(ans, x);
        }
        return ans;
    }
};

D. 使数组连续的最少操作数#

排序 + 二分搜索. 重复元素是一定需要操作的,因此排序后去重,记录新的长度m. 对于任意的$i < m $,通过二分查找找到第一个大于$nums[i] + n - 1$的值. 区间[left, right]中的值即为不需要操作的值,区间外的值都需要操作.

  • 时间复杂度: $O(nlgn)$
  • 空间复杂度: $O(1)$
class Solution {
public:
    int minOperations(vector<int>& nums) {
        sort(begin(nums), end(nums));
        int n = nums.size();
        int m = remove_duplicate(nums);
        int ans = n;
        int r = 0; 
        for (int i = 0; i < m; i++) {
            r = binary_search(nums, i, m, n - 1 + nums[i]);
            // cout << r << endl;
            ans = min(ans, n - (r - i));
        }
        return ans;
    }

    int remove_duplicate(vector<int>& nums) {
        int i = 0;
        int n = nums.size();
        for (int j = 0; j < n; j++) {
            if (nums[j] == nums[i]) continue;
            nums[++i] = nums[j];
        }
        return i + 1;
    }

    int binary_search(vector<int>& nums, int left, int right, int target) {
        // cout << "[" << left << "," << right << "] target=" << target << endl;
        while (left < right) {
            int mid = ((right - left) >> 1) + left;
            if (nums[mid] < target) {
                left = mid + 1;
            }else if (nums[mid] == target) {
                return mid + 1;
            }else {
                right = mid;
            }
        }
        return left;
    }
};

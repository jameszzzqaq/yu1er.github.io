准备#

repo ceph源包制作和安装#

  1. 安装工具包
  • 机器要求能够连外网
yum -y install epel-release 
yum -y install yum-utils 
yum -y install createrepo 
  1. 添加ceph octopus配置 (octopus版本为15.2.15) vim /etc/yum.repos.d/ceph.repo 内容如下：
[ceph]
name=ceph
baseurl=http://mirrors.aliyun.com/ceph/rpm-15.2.15/el7/x86_64/
gpgcheck=0
[ceph-noarch]
name=cephnoarch
baseurl=http://mirrors.aliyun.com/ceph/rpm-15.2.15/el7/noarch/
gpgcheck=0
[ceph-source]
name=cephsource
baseurl=http://mirrors.aliyun.com/ceph/rpm-15.2.15/el7/x86_64/
gpgcheck=0
[ceph-radosgw]
name=cephradosgw
baseurl=http://mirrors.aliyun.com/ceph/rpm-15.2.15/el7/x86_64/
gpgcheck=0
  1. 制作repo
yum clean all
yum repolist

# 查看ceph-deploy版本
yum list all | grep ceph-deploy

# 下载ceph包和依赖
mkdir local_repo

# 以防万一把ssh、ntp、crontabs、python-setuptools也下载一下
# repotrack openssh openssh-clients openssh-server ntp ntpupdate crontabs python-setuptools -p /root/local_repo
repotrack ceph-deploy-2.0.1 -p /root/local_repo
repotrack ceph-15.2.15 ceph-mgr-15.2.15 ceph-mon-15.2.15 ceph-mds-15.2.15 ceph-osd-15.2.15 ceph-fuse-15.2.15 ceph-radosgw-15.2.15 -p /root/local_repo

# 生成仓库
createrepo -v /root/local_repo

# 压缩打包
tar -zcvf local_repo.tar.gz -C /root local_repo

4. 到指定机器上安装(强制安装）(参考下一步修改源来安装) tar -zxvf local_repo.tar.gz cd ceph-15.2.15 # rpm -ivh *.rpm rpm -ivh *.rpm --nodeps --force

从自制的源安装#

  1. 解压缩
tar -zxvf local_repo.tar.gz
  1. 备份原有的源,并修改为自制的源
cd /etc/ && mv yum.repos.d/ yum.repos.d.back/
mkdir yum.repos.d && cd yum.repos.d && vi local.repo

内容为：

[local]
name=local
baseurl=file:///root/local_repo 
enabled=1
gpgcheck=0
  • baseurl为自己的源的路径
  1. 更新
yum clean all && yum repolist

配置主机名#

# 1. 修改每台主机的主机名
hostnamectl --static set-hostname client
# 依次修改
# hostnamectl --static set-hostname ceph1
# hostnamectl --static set-hostname ceph2
# hostnamectl --static set-hostname ceph3

# 2. 添加hosts解析
echo -e '192.168.3.11\tceph1\n192.168.3.12\tceph2\n192.168.3.13\tceph3\n192.168.3.20\tclient' >> /etc/hosts

配置ntp#

# 1. 安装ntp
yum install -y ntp ntpdate
# 2. 修改配置
cd /etc && mv ntp.conf ntp.conf.bak && vi /etc/ntp.conf
# -2.1. 对于ceph1,内容为：
####################
restrict 127.0.0.1
restrict ::1
restrict 192.168.3.0 mask 255.255.255.0
server 127.127.1.0
fudge 127.127.1.0 stratum 8
####################
# 192.168.3.0为**ceph1**的网段

# -2.2. 对于其他机器，内容为
server 192.168.3.11 
# 192.168.3.11为ceph1的ip

# 也可用命令
# 对于ceph1:
# cd /etc && mv ntp.conf ntp.conf.bak && echo -e 'restrict 127.0.0.1\nrestrict ::1\nrestrict 192.168.3.0 mask 255.255.255.0\nserver 127.127.1.0\nfudge 127.127.1.0 stratum 8' > /etc/ntp.conf
# 其他机器：
# cd /etc && mv ntp.conf ntp.conf.bak && echo -e 'server 192.168.3.11' > /etc/ntp.conf

# 3. 启动ntp
# -3.1. 只在ceph1启动
systemctl start ntpd
systemctl enable ntpd
# systemctl status ntpd

# -3.2. 除ceph1节点外，其余节点与ceph1同步
ntpdate ceph1

# -3.3. 除ceph1节点外，其余节点写入硬时钟
hwclock -w

# -3.4. 除ceph1节点外，其余节点添加crontabs，定时同步
yum install -y crontabs
chkconfig crond on
systemctl start crond
crontab -e
# -3.5. 写入以下内容，每隔10分钟和ceph1同步一次
##############
*/10 * * * * /usr/sbin/ntpdate 192.168.3.11
##############

配置ssh#

# 1. 安装openssh openssh-clients openssh-server
yum install -y openssh openssh-clients openssh-server
systemctl start sshd

# 2. 在ceph1产生key并分发
ssh-keygen -t rsa
for i in {1..3}; do ssh-copy-id ceph$i; done
ssh-copy-id client

关闭防火墙#

systemctl stop firewalld
systemctl disable firewalld
systemctl status firewalld

关闭selinux#

# 1. 两种方式
# 临时关闭,重启失效
setenforce 0
# 永久关闭，重启生效
vi /etc/selinux/config
# 修改SELINUX=disabled

安装部署#

安装ceph软件#

# 1. 所有节点安装ceph、ceph-radosgw
yum install -y ceph ceph-radosgw

# 2. ceph1节点额外安装ceph-deploy
yum -y install ceph-deploy

# 3. 查看ceph版本
ceph -v

部署Mon节点#

  • 只需要在ceph1上操作
# 1. 创建集群 (TODO: yum -y install python-setuptools)
cd /etc/ceph
ceph-deploy new ceph1 ceph2 ceph3

# 2. 修改配置文件
vi /etc/ceph/ceph.conf

# 内容如下
#############################
[global]
fsid = f6b3c38c-7241-44b3-b433-52e276dd53c6
mon_initial_members = ceph1, ceph2, ceph3
mon_host = 192.168.3.11,192.168.3.12,192.168.3.13
auth_cluster_required = cephx
auth_service_required = cephx
auth_client_required = cephx
public_network = 192.168.3.0/24
cluster_network = 192.168.4.0/24

[mon]
mon_allow_pool_delete = true
################################

# 3. 初始化监视器并收集密钥
ceph-deploy mon create-initial

# 4. n 将ceph.client.admin.keyring拷贝到各个节点上。
ceph-deploy --overwrite-conf admin client ceph1 ceph2 ceph3

  • 配置节点命令必须在/etc/ceph路径下执行

  • 网络隔离，cluster_network用于存储节点之间交互，public_network用于存储节点和外部节点交互

  • 警告：mons are allowing insecure global_id reclaim ceph config set mon auth_allow_insecure_global_id_reclaim false

部署Mgr节点#

# 1. 部署mgr
ceph-deploy mgr create ceph1 ceph2 ceph3

部署Osd#

# 在ceph1节点部署一个osd(需要在每台机器部署多个osd)
# wal,db和data分开部署
# ceph-deploy osd create ceph1 --data /dev/sda --block-wal /dev/nvmea0 --block-db /dev/nvmea1

# wal,db和data一起部署
ceph-deploy osd create ceph1 --data /dev/sda

  • docker-compose集群搭建#

version: "3.3"

services:
  ceph1:
    image: centos:7
    container_name: ceph1
    hostname: ceph1
    stdin_open: true
    tty: true
    privileged: true
    networks:
      ceph-net:
        ipv4_address: 192.168.3.11
    command: ['/usr/sbin/init']

  ceph2:
    image: centos:7
    container_name: ceph2
    hostname: ceph2
    stdin_open: true
    tty: true
    privileged: true
    networks:
      ceph-net:
        ipv4_address: 192.168.3.12
    command: ['/usr/sbin/init']

  ceph3:
    image: centos:7
    container_name: ceph3
    hostname: ceph3
    stdin_open: true
    tty: true
    privileged: true
    networks:
      ceph-net:
        ipv4_address: 192.168.3.13
    command: ['/usr/sbin/init']

  client:
    image: centos:7
    container_name: ceph-client
    hostname: client
    stdin_open: true
    tty: true
    privileged: true
    networks:
      ceph-net:
        ipv4_address: 192.168.3.20
    command: ['/usr/sbin/init']

networks:
  ceph-net:
    driver: bridge
    ipam:
      driver: default
      config:
        - subnet: 192.168.3.0/24
